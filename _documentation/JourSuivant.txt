# ✅ 5 — Modifier downloadChunk()

Dans CopainImporter

### nouvelle version

```php
public function downloadChunk()
{
    $ref = $this->competitionId;

    $zipPath =
        WRITEPATH .
        'import/' .
        $ref .
        '.zip';

    if (!is_dir(WRITEPATH.'import'))
        mkdir(WRITEPATH.'import',0777,true);


    /*
    -----------------------
    get zip url
    -----------------------
    */

    $zipInfo =
        $this->client->generateZip($ref,1);

    if (
        !$zipInfo
        || empty($zipInfo['zip_photos'])
    ) {
        return true;
    }

    $url =
        $zipInfo['zip_photos'];


    /*
    -----------------------
    remote size
    -----------------------
    */

    $size =
        $this->client
        ->getRemoteFileSize($url);


    $localSize =
        file_exists($zipPath)
        ? filesize($zipPath)
        : 0;


    log_message(
        'debug',
        "SIZE $localSize / $size"
    );


    /*
    -----------------------
    done ?
    -----------------------
    */

    if ($localSize >= $size) {

        log_message(
            'debug',
            'DOWNLOAD DONE'
        );

        return true;
    }


    /*
    -----------------------
    download chunk
    -----------------------
    */

    $fp =
        fopen($zipPath,'a');

    $curl = curl_init();

    curl_setopt_array($curl,[

        CURLOPT_URL => $url,

        CURLOPT_FILE => $fp,

        CURLOPT_RANGE =>
            $localSize.'-'.
            ($localSize + 50*1024*1024),

        CURLOPT_FOLLOWLOCATION => true,

        CURLOPT_SSL_VERIFYPEER => false,
        CURLOPT_SSL_VERIFYHOST => false,

        CURLOPT_COOKIEJAR =>
            WRITEPATH.'copain_cookie.txt',

        CURLOPT_COOKIEFILE =>
            WRITEPATH.'copain_cookie.txt',

    ]);

    curl_exec($curl);

    curl_close($curl);

    fclose($fp);


    return false;
}
```

# ✅ 1 — FIX MAJEUR : download progress + chunk réel

## 🔧 remplacer ENTIEREMENT dans CopainImporter

```php
public function downloadChunk()
{
    $ref = $this->competitionId;

    if (!$ref) return true;

    $zipPath =
        WRITEPATH . 'imports/' . $ref . '.zip';

    $sizeFile =
        WRITEPATH . "imports/$ref/size.txt";

    /*
    -----------------------
    url + size
    -----------------------
    */

    if (!file_exists($sizeFile)) {

        $zipInfo =
            $this->client->generateZip(
                $ref,
                1
            );

        if (
            !$zipInfo
            || empty($zipInfo['zip_photos'])
        ) {
            log_message('error', 'NO ZIP URL');
            return true;
        }

        $url = $zipInfo['zip_photos'];

        $size =
            $this->client
                ->getRemoteFileSize($url);

        file_put_contents(
            $sizeFile,
            json_encode([
                'url' => $url,
                'size' => $size
            ])
        );
    }

    $info =
        json_decode(
            file_get_contents($sizeFile),
            true
        );

    $url = $info['url'];
    $size = $info['size'];

    if (!$size) {
        log_message('error', 'SIZE ERROR');
        return true;
    }

    $localSize =
        file_exists($zipPath)
        ? filesize($zipPath)
        : 0;

    /*
    -----------------------
    done
    -----------------------
    */

    if ($localSize >= $size) {

        log_message('debug', 'DOWNLOAD DONE');

        return true;
    }

    /*
    -----------------------
    chunk 5MB
    -----------------------
    */

    $chunk = 5 * 1024 * 1024;

    $this->client->downloadRange(
        $url,
        $zipPath,
        $localSize,
        $localSize + $chunk - 1
    );

    return false;
}
```

# ✅ 2 — ajouter downloadRange dans CopainClient

AJOUTER :

```php
public function downloadRange(
    $url,
    $dest,
    $start,
    $end
) {

    $fp = fopen(
        $dest,
        $start == 0 ? 'wb' : 'ab'
    );

    $curl = curl_init($url);

    curl_setopt_array($curl, [

        CURLOPT_FILE => $fp,

        CURLOPT_RANGE =>
        $start . "-" . $end,

        CURLOPT_COOKIEJAR =>
        $this->cookie,

        CURLOPT_COOKIEFILE =>
        $this->cookie,

        CURLOPT_SSL_VERIFYPEER => false,
        CURLOPT_SSL_VERIFYHOST => false,

        CURLOPT_TIMEOUT => 0,
        CURLOPT_FOLLOWLOCATION => true,

    ]);

    curl_exec($curl);

    curl_close($curl);
    fclose($fp);
}
```
